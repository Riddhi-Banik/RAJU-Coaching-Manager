function vectorize_student_data(JsonDt)
{
    const rows = Object.keys(JsonDt).length;
/** @type {{cls: string, batch: string, roll: string, name: string, gender: string, v_balance: number|null, mobile_no: string}[]} */

    };

    return lst
};



/** @type {HTMLTableRowElement} */
var selected_element = null;
/** 
@param {HTMLTableRowElement}  rowElem
*/

function select_from_table(rowElem)
{
    if (selected_element)
    {
       selected_element.classList.remove('selected'); 
    }
    if (selected_element === rowElem)
    {
        //deselect
        selected_element = null;
    }
    else 
    {
        //select
        selected_element = rowElem;
        selected_element.classList.add('selected');
    }
};

var form_tab_opened = document.getElementById(('Add' + 'Content'));

/** 
 * @param {tab == string;} -> void
 */
function select_tab(tabID)
{
    if (!form_tab_opened) {
        form_tab_opened = document.getElementById('AddContent');
        form_tab_opened_nav = document.getElementById('AddNav');
    }
    /** @type {HTMLElement}*/
    let tab = document.getElementById((tabID + 'Content')); 
    let nav = document.getElementById((tabID + 'Nav'));
    if (tab === form_tab_opened) // do nothin
    {
        return
    } 
    else // swap display target
    {
        form_tab_opened.style.display = 'none';
        form_tab_opened_nav.classList.remove('selected');

        form_tab_opened = tab;
        form_tab_opened_nav = nav;

        form_tab_opened.style.display = 'block';
        form_tab_opened_nav.classList.add('selected'); 

    }


}
